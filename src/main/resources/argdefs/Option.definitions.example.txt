## Example long-form option definitions
##

[p path]
name=path
description=The path to search
type=PATH

[server]
name=server
description=Server name or IP address
type=STRING
validation=STRING.not.empty

[port]
name=port
description=Server port
type=INTEGER
validation.INTEGER.positive

[file]
name=file
description=input file
type=PATH
validation=PATH.exists
validation=PATH.readable
